163  int execute(SQLITE3_QUERY &query) {
164  exec_lock->lock(); // lock exec
165 
166  // check if database connection is open
167  if (!*db) {
168  error_no = 4;
169  exec_lock->unlock(); // unlock exec
170 
171  return 1;
172  }
173 
174  // get query from SQLITE3_QUERY
175  std::string prepared_query;
176  try {
177  prepared_query = query.bind().bound_query;
178  } catch (std::out_of_range &e) {
179  error_no = 3;
180  exec_lock->unlock(); // unlock exec
181 
182  return 1;
183  }
184 
185  // clear result and column vector
186  result->clear();
187  column_name->clear();
188 
189  // run query
190  auto data_pack = Callback_Data(column_name, result);
191  int rc = sqlite3_exec(*db, prepared_query.c_str(), &exec_callback, &data_pack, err_msg.get());
192  if (rc != SQLITE_OK) { // check for error
193  // copy error message and free memory
194  err_msg_str = std::string(*err_msg);
195  sqlite3_free(*err_msg);

455  static int exec_callback(void *ptr, int argc, char *argv[], char *col_name[]) {
456  auto *data = reinterpret_cast<Callback_Data *>(ptr);
457 
458  // record column name if needed
459  if (data->col->empty()) {
460  for (int i = 0; i < argc; ++i) {
461  data->col->push_back(std::string(col_name[i] ? col_name[i] : "NULL"));
462  }
463  }
464 
465  // get result
466  SQLITE_ROW_VECTOR row;
467  for (int i = 0; i < argc; i++) {
468  row.push_back(std::string(argv[i] ? argv[i] : "NULL"));
469  }
470 
471  // push SQLITE_ROW_VECTOR to result vector
472  data->rows->push_back(row);
473 
474  return 0;
475  }
